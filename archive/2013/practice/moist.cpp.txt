#include <iostream>
#include <string>
using namespace std;

string solve() {
    int n;
    cin >> n;
    string largest;
    cin.ignore(1); // ignore one char
    getline(cin, largest);
    int cost = 0;
    for (int j = 1; j < n; j++) {
        string s;
        getline(cin, s);
        if (s < largest) {
            ++cost;
        } else {
            largest = s;
        }
    }
    return to_string(cost) + "\n";
}

int main() {
#ifdef PLAYGROUND_LOCAL
    FILE *fre = freopen("kickstart.in", "r", stdin);
#endif
    int t;
    cin >> t;
    for (int i = 1; i <= t; i++) {
        cout << "Case #" + to_string(i) + ": " + solve();
    }
#ifdef PLAYGROUND_LOCAL
    fclose(fre);
#endif
}
